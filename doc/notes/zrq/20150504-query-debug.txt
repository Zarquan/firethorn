    JOIN query fails because it is treated as DISTRIBUTED

        SELECT
            TOP 10
            S.sourceID as ID,
            S.ra as ra,
            S.dec as decl
        FROM
            atlasSource as S
        JOIN
            atlasSourceXwise_allskysc as X on X.masterObjID = S.sourceID
        JOIN
            wise_allskysc as W on X.slaveObjID = W.cntr

            "http://localhost:8093/firethorn/jdbc/resource/4"
            ],

# -----------------------------------------------------
# Example template for a development branch.
# http://redmine.roe.ac.uk/projects/wva/wiki/20140108MercurialBranches
# http://wfau.metagrid.co.uk/code/firethorn/graph/9aadad0e6b82
#

# -----------------------------------------------------
# Set the branch name, based on pattern 
#[user@desktop]

    newbranch=[version]-[who]-[description]

    newbranch=1.9.3-stv-datatable-controller
    newbranch=1.9.3-zrq-ogsa-delay
    newbranch=1.9.4-zrq-example


# -----------------------------------------------------
# Create the new branch.
#[user@desktop]

    source "${HOME:?}/firethorn.settings"
    pushd "${FIRETHORN_CODE:?}"

        #
        # Check we have the most recent version.
        hg pull
        hg update

        #
        # Create the new branch.
        hg update default
        hg branch "${newbranch:?}"

        #
        # Set the project version.
        ./bin/version.sh "${newbranch:?}"

        #
        # Commit and push the new branch.
        hg commit -m "Creating [branch-${newbranch:?}]"
        hg push --new-branch

        #
        # Re-build everything.
        mvn -P all clean install
        mvn -P all eclipse:eclipse

    popd

# -----------------------------------------------------
# Add the new code ....
#[user@desktop]

    source "${HOME:?}/firethorn.settings"
    pushd "${FIRETHORN_CODE:?}"

        #
        # .....
        # .....
        # .....
        #

    popd

# -----------------------------------------------------
# Close the branch.
#[user@desktop]

    source "${HOME:?}/firethorn.settings"
    pushd "${FIRETHORN_CODE:?}"

        hg update "${newbranch:?}"
        hg commit --close-branch -m "Closing [branch-${newbranch:?}]"
        hg push

    popd

# -----------------------------------------------------
# Merge the branch into default.
#[user@desktop]

    source "${HOME:?}/firethorn.settings"
    pushd "${FIRETHORN_CODE:?}"

        hg pull
        hg update

        #
        # Fix any conflicts ...
        #

        hg update default
        hg merge "${newbranch:?}"

        #
        # Fix any conflicts ...
        #

        hg commit -m "Merging [branch-${newbranch:?}] into [default]"
        hg push

    popd

# -----------------------------------------------------
# Increment the project version.
#[user@desktop]

    newversion=1.9.5

    source "${HOME:?}/firethorn.settings"
    pushd "${FIRETHORN_CODE:?}"

        hg update default

        #
        # Set the project version.
        ./bin/version.sh "${newversion:?}"

        #
        # Tag the new version.
        hg commit -m "version-${newversion:?}"
        hg tag    -f "version-${newversion:?}"
        hg push

        #
        # Re-build everything.
        mvn -P all clean install
        mvn -P all eclipse:eclipse

        #
        # Push to our public repositiry.
        ./bin/rsync-repo.sh

    popd

#
# Adding the following to our Fedora Dockerfile sigificantly reduced the size of our images.


vi docker/fedora/8/Dockerfile

    #
    # Declare the system cache as a volume.
    VOLUME /var/cache/

    #
    # Clear the cache.
    RUN yum -y clean all
